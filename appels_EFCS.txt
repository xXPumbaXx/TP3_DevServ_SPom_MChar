## Consulter un film en particulier
query {
  Film(id: 1) {
    title
    critics {
      id
      score
      comment
    }
  }
}

## Consultation de tous les acteurs d’un certain film
query {
  Film(id: 1) {
    title
    actors {
      last_name
      first_name
      birthdate
    }
  }
}

## Recherche de films - Cas 1 : Sans critère avec pagination (demandez la page 1)
query {
  SearchFilms(page: 1) {
    data {
      id
      title
      release_year
      length
      description
      rating
    }
    paginatorInfo {
      currentPage
      lastPage
      total
    }
  }
}

## Recherche de films - Cas 2 : Avec le mot-clé "AL" (sur titre ou descritpion) avec pagination (demandez la page 1)
query {
  SearchFilms(title: "%AL%", page: 1) {
    data {
      id
      title
      release_year
      length
      description
      rating
    }
    paginatorInfo {
      currentPage
      lastPage
      total
    }
  }
}

## Recherche de films - Cas 3 : Avec le mot-clé "AL" (sur titre ou descritpion), minimum année 2006, entre 120 et 150 minutes avec pagination (demandez la page 1)
query {
  SearchFilms(title: "%AL%", release_year: 2006, min: 120, max: 150, page: 1) {
    data {
      id
      title
      release_year
      length
      description
      rating
    }
    paginatorInfo {
      currentPage
      lastPage
      total
    }
  }
}
